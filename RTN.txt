fetch:
    MA<-PC; C<-PC+2
    MD<-MatMA
    IR<-MD
add:
    A<-Rsrc
    B<-Rdest
    C<-A+B
    Rdest<-C
xor:
    A<-Rsrc
    B<-Rdest
    C<-A^B
    Rdest<-C
sub:
    A<-Rsrc
    B<-Rdest
    C<-A-B
    Rdest<-C
and:
    A<-Rsrc
    B<-Rdest
    C<-A&B
    Rdest<-C
irmov:
    Rdest<-src
rrmov:
    Rdest<-Rsrc
jmp:
    PC<-src
push:
    MA<-Rsp
    MatMA<-Rsrc
pop:
    MA<-Rsp
    Rsrc<-MatMA
call:
    MA<-Rsp
    PC<-src
ret:
    Rsp<-MatMA
